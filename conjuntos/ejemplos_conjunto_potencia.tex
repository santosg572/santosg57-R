\begin{frame}{Frame Title}
    \frametitle{Conjuntos}

\textbf{Ejemplos.}

\hfill

\begin{enumerate}
   \item Si $A = \{ 1,2,3 \}$ entonces $2^A = \{ \O, \{ 1 \}, \{ 2 \}, \{ 3 \}, \{ 1,2 \}, \{ 1,3 \}, \{ 2,3 \}, \{ 1,2,3 \} \}$
   \item  Si $A = \O$ entonces el único subconjunto de A es $\O$ así que $2^{\O} = { \O }$
\end{enumerate}

\end{frame}
