  print('VERDADERO')
}

  print('VERDADERO')
}

FOR

s = 0
for (i in 1:100){
  s = s+i
}
print(s)

for (i in 1:100){
  if (i %% 2 == 0)
    print(i)
}

while

s = 0
i = 1

while (i <=100){
  s = s+i^2
  i = i+1
}

print(s)
